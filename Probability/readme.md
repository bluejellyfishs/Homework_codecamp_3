  A = C => False
  B ~ D => True
  D ⊆ C => False
  C ⊆ D => True
  A ≠ B => True

  A = {1,2,3} B = {3,4,5,6,7}

  เขียนสมาชิกเซตต่อไปนี้
  A ∪ B ∪ C => {1, 2, 3, 4, 5, 6, 7, 9}
  A ∩ B ∩ C => {Ø}

  (A-B)c => {3, 4, 5, 6, 7, 8, 9, 10}
  เปรียบเทียบระหว่าง 2 เซตนี้
  (A ∩ C) = Ac ∪ Cc => False
  (A ∩ B) ∪ C = (A U C) ∩ (B U C) => True
